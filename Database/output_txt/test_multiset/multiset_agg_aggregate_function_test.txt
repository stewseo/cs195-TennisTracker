select
  "sakila"."film"."title",
  multiset(
    select "actor"."FIRST_NAME", "actor"."LAST_NAME"
    from "FILM_ACTOR"
      join "actor"
        on "actor"."ACTOR_ID" = "FILM_ACTOR"."ACTOR_ID"
    where "FILM_ACTOR"."FILM_ID" = "sakila"."film"."film_id"
  ) "actors",
  multiset(
    select "category"."name"
    from "film_category"
      join "category"
        on "category"."category_id" = "film_category"."category_id"
    where "film_category"."film_id" = "sakila"."film"."film_id"
  ) "categories",
  multiset(
    select
      "sakila"."customer"."first_name",
      "sakila"."customer"."last_name",
      multiset_agg("payment"."payment_date") "payments",
      sum("payment"."amount") "total"
    from "payment"
      join "sakila"."customer"
        on "payment"."customer_id" = "sakila"."customer"."customer_id"
      join "rental"
        on "payment"."rental_id" = "rental"."rental_id"
      join "inventory"
        on "rental"."inventory_id" = "inventory"."inventory_id"
    where "inventory"."film_id" = "sakila"."film"."film_id"
    group by "sakila"."customer"."customer_id", "sakila"."customer"."first_name", "sakila"."customer"."last_name"
  ) "customers"
from "sakila"."film"
where "sakila"."film"."title" like 'B%'
order by "sakila"."film"."film_id"
limit 10

  `sakila`.`film`.`title`,

      where `FILM_ACTOR`.`FILM_ID` = `sakila`.`film`.`film_id`
    ) as `t`
  ) as `actors`,
  (
    select coalesce(

        '[]',

      select `category`.`name` as `v0`

  ) as `categories`,

  ) as `customers`
